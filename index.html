Q6)Find the Winner

Sol: 
        var Game_Winner = (S) => 
        {
          let N=S.length;
          countA=0;
          countD=0;
          for(i=0;i < N;i++)
          {
            if(S[i]==='A')
            {
              countA++;
            }
            if(S[i]==='D')
            {
              countD++;
            }
          }
          if(countA==countD)
          {
            return 'Draw';
          }
          else
          {
            if(countA>countD)
            {
              return 'Aditya' ;
            }
            else
            {
              return 'Danish';
            }
          }
        }
